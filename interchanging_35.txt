#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    int n;
    cin >> n;  
    
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        cin >> arr[i];  
    
    }
    int min = 0, max = 0;
    for (int i = 1; i < n; i++) {
        if (arr[i] < arr[min]) {
            min = i;
        }
        if (arr[i] > arr[max]) {
            max = i;
        }
    }
    
  
    int temp = arr[min];
    arr[min] = arr[max];
    arr[max] = temp;
    

    for (int i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }
    
    return 0;
}
