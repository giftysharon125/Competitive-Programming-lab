#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>


    /* Enter your code here. Read input from STDIN. Print output to STDOUT */    int factorial(int n) {
    if (n == 0 || n == 1) {
        return 1;  
    }
    return n * factorial(n - 1);  
    
}

int main() {
    int num;
    scanf("%d", &num);  
    
    printf("%d\n", factorial(num)); 
    
    return 0;
}
